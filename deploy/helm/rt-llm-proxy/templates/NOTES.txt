rt-llm-proxy is installed.

1. Build and load the image (Minikube example):
   docker build -t {{ .Values.image.repository }}:{{ .Values.image.tag }} .
   minikube image load {{ .Values.image.repository }}:{{ .Values.image.tag }}

2. Open the demo UI:
{{- if eq .Values.service.type "NodePort" }}
   export NODE_PORT=$(kubectl get svc {{ include "rt-llm-proxy.fullname" . }} -n {{ .Release.Namespace }} -o jsonpath='{.spec.ports[0].nodePort}')
   echo "http://$(minikube ip):${NODE_PORT}/demo/"
{{- else }}
   kubectl port-forward svc/{{ include "rt-llm-proxy.fullname" . }} 8080:{{ .Values.service.port }} -n {{ .Release.Namespace }}
   echo "http://127.0.0.1:8080/demo/"
{{- end }}

Redis (rate limiting): {{ include "rt-llm-proxy.redis.addr" . }}
